# views/auswahl_seite.py
import logging
logger = logging.getLogger(__name__)

from PyQt5 import uic
from PyQt5.QtWidgets import QWidget
from pathlib import Path

class AuswahlSeite(QWidget):
    def __init__(self, parent=None):
        super().__init__(parent)
        ui_path = Path(__file__).parent / "auswahl_seite.ui"
        uic.loadUi(str(ui_path), self)

        # Buttons mit Methoden im Parent verbinden
        self.btn_to_side_fu_heu.clicked.connect(lambda: parent.zeige_heu_futter() if parent else None)
        self.btn_to_side_fu_heulage.clicked.connect(lambda: parent.zeige_heulage_futter() if parent else None)
        self.btn_to_side_fu_laden.clicked.connect(lambda: parent.zeige_futter_laden() if parent else None)
        self.btn_to_side_settings.clicked.connect(lambda: parent.zeige_einstellungen() if parent else None)
